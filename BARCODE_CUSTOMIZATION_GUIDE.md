// ============================================
// السطر: 28 - 36 في الملف

JsBarcode(svg, barcodeValue, {
  format: 'CODE128',           // ⚙️ صيغة الباركود (CODE128, EAN13, UPC, etc)
  width: 2,                    // ⚙️ عرض الخطوط (1-3 بكسل)
  height: 80,                  // ⚙️ ارتفاع الباركود (بالبكسل)
  displayValue: true,          // ⚙️ عرض النص تحت الباركود (true/false)
  fontSize: 16,                // ⚙️ حجم الخط للنص (بكسل)
  margin: 10,                  // ⚙️ الهامش حول الباركود (بكسل)
  background: '#ffffff',       // ⚙️ لون الخلفية (أبيض)
  lineColor: '#000000'         // ⚙️ لون الخطوط (أسود)
});

/**
 * أمثلة على التخصيص:
 * 
 * 1. باركود أكبر للطابعات الحرارية:
 *    width: 2.5
 *    height: 120
 *    fontSize: 14
 * 
 * 2. باركود صغير للملصقات:
 *    width: 1.5
 *    height: 50
 *    fontSize: 10
 * 
 * 3. بدون نص:
 *    displayValue: false
 * 
 * 4. باركود ملون (غير شائع):
 *    lineColor: '#d32f2f'  // أحمر
 */
